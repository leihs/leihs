100:       attribute_names = self.model.columns.collect{ |c| c.name.to_sym }.sort_by { |c| c.to_s }
101:       association_column_names = self.model.reflect_on_all_associations.collect{ |a| a.name.to_sym }.sort_by { |c| c.to_s }
102:       @columns = ActiveScaffold::DataStructures::Columns.new(self.model, attribute_names + association_column_names)
103: 
104:       # and then, let's remove some columns from the inheritable set.
105:       @columns.exclude(*self.class.ignore_columns)
106:       @columns.exclude(*@columns.find_all { |c| c.column and (c.column.primary or c.column.name =~ /(_id|_count)$/) }.collect {|c| c.name})
107:       @columns.exclude(*self.model.reflect_on_all_associations.collect{|a| "#{a.name}_type""#{a.name}_type" if a.options[:polymorphic]}.compact)
108: 
109:       # inherit the global frontend
110:       @frontend = self.class.frontend
111:       @theme = self.class.theme
